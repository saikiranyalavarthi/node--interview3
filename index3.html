3.Create a middleware which will be applicable to all the routes 
    
    const express = require('express');
const app = express();
const mid=(req,res,next)=>{
    console.log("i am middleware1 globally");
    next();
}

app.use(mid);

app.get('/',mid,(req,res)=>{
    res.send('hello from home');

});
app.get('/about', mid,(req,res)=>{
    res.send(' I am from About page')
});
app.get('/contact',mid,(req,res)=>{
    res.send(' I am from Contact Page')
});
app.get('/something',mid,(req,res)=>{
    res.send('I am from Something')
});

app.listen(8000,()=>{
    console.log('server is running');
})
